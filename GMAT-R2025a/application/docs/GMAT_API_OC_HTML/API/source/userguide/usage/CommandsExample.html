Propagation Command¶

Listing 19 Setting up and running a Propagate command¶
 1from load_gmat import gmat
 2
 3# Core resources used
 4sat  = gmat.Construct("Spacecraft", "Sat")
 5sat.SetField("DateFormat", "UTCGregorian")
 6sat.SetField("Epoch", "02 Jan 2023 12:00:00.000")
 7sat.SetField("X", 0)
 8sat.SetField("Y", -3100)
 9sat.SetField("Z", -11591)
10sat.SetField("VX", 5.155)
11sat.SetField("VY", 0.0)
12sat.SetField("VZ", 0.0)
13
14fm   = gmat.Construct("ForceModel", "Dynamics")
15epm  = gmat.Construct("PointMassForce", "EPM")
16fm.AddForce(epm)
17
18prop = gmat.Construct("Propagator", "Prop")
19prop.SetField("FM", "Dynamics")
20
21propagate = gmat.Command("Propagate", "Prop(Sat) {Sat.Periapsis}")
22
23gmat.Initialize()
24
25gmat.Execute()

Targeting Command¶

Listing 20 Setting up and running a Propagate command¶
 1from load_gmat import gmat
 2
 3# Core resources used
 4sat  = gmat.Construct("Spacecraft", "Sat")
 5sat.SetField("DateFormat", "UTCGregorian")
 6sat.SetField("Epoch", "10 Feb 2023 12:00:00.000")
 7sat.SetField("DispayStateType", "Keplerian")
 8sat.SetField("SMA", 6900)
 9sat.SetField("ECC", 0.005)
10sat.SetField("INC", 28.5)
11sat.SetField("RAAN", 45)
12sat.SetField("AOP", 90.0)
13sat.SetField("TA", 180.0)
14
15fm   = gmat.Construct("ForceModel", "Dynamics")
16epm  = gmat.Construct("PointMassForce", "EPM")
17fm.AddForce(epm)
18
19prop = gmat.Construct("Propagator", "Prop")
20prop.SetField("FM", "Dynamics")
21
22burn = gmat.Construct("ImpulsiveBurn", "Burn")
23dc = gmat.Construct("DifferentialCorrector", "DC")
24
25# First propagate so we start at perigee
26gmat.Command("Propagate", "Prop(Sat) {Sat.Periapsis}")
27
28# Target commands
29gmat.Command("Target", "DC")
30# Note that commands are added to the branch in the Target command
31# until the EndTarget command is added
32gmat.Command("Vary", "DC(Burn.Element1 = 0.5, {Perturbation = 0.0001})")
33gmat.Command("Maneuver", "Burn(Sat)")
34gmat.Command("Propagate", "Prop(Sat) {Sat.Apoapsis}")
35gmat.Command("Achieve", "DC(Sat.Earth.RMAG = 42165.0, {Tolerance = 0.1})")
36gmat.Command("EndTarget")
37
38gmat.Initialize()
39
40# Execute() runs the entire sequence to completion
41gmat.Execute()
